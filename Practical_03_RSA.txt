# Simple RSA Program

# Choose two prime numbers
p = 3
q = 11

# Calculate n
n = p * q

# Calculate phi
phi = (p - 1) * (q - 1)

# Choose e
e = 7

# Find d such that (d * e) % phi = 1
d = 3

# Original message
message = 9

# Encryption
cipher = (message ** e) % n

# Decryption
plain = (cipher ** d) % n

print("Value of n        :", n)
print("Value of phi      :", phi)
print("Public Key (e,n)  :", (e, n))
print("Private Key (d,n) :", (d, n))
print("Original Message  :", message)
print("Encrypted Message :", cipher)
print("Decrypted Message :", plain)

// Save as: practical3.cpp

#include <iostream>
#include <cmath>

using namespace std;

int main()
{
    int p = 3;
    int q = 11;

    int n = p * q;
    int phi = (p - 1) * (q - 1);

    int e = 7;
    int d = 3;

    int message = 9;

    int cipher = (int)pow(message, e) % n;
    int plain  = (int)pow(cipher, d) % n;

    cout << "Value of n        : " << n << endl;
    cout << "Value of phi      : " << phi << endl;
    cout << "Public Key (e,n)  : (" << e << "," << n << ")" << endl;
    cout << "Private Key (d,n) : (" << d << "," << n << ")" << endl;
    cout << "Original Message  : " << message << endl;
    cout << "Encrypted Message : " << cipher << endl;
    cout << "Decrypted Message : " << plain << endl;

    return 0;
}

Q1. What is RSA?

Q2. How many keys are used in RSA?
