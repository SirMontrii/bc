#include <iostream>
#include <iomanip>
#include <cstring>
#include <openssl/sha.h>

using namespace std;

int main()
{
    string message = "Blockchain Developer";

    unsigned char hash[SHA256_DIGEST_LENGTH];

    SHA256((unsigned char*)message.c_str(),
           message.length(),
           hash);

    cout << "Original Message : " << message << endl;
    cout << "SHA-256 Digest   : ";

    for(int i = 0; i < SHA256_DIGEST_LENGTH; i++)
    {
        cout << hex
             << setw(2)
             << setfill('0')
             << (int)hash[i];
    }

    cout << endl;

    return 0;
}

-------------------------
VIVA QUESTIONS
-------------------------

Q1. What is SHA-256?
Ans: SHA-256 is a cryptographic hash function that
produces a 256-bit fixed length hash value.

Q2. Can SHA-256 be decrypted?
Ans: No, it is a one-way function.

Q3. Why is SHA-256 used in blockchain?
Ans: It is used for block hashing, security,
and tamper detection.

Q4. If same input is given again, same output comes?
Ans: Yes, SHA-256 is deterministic.
