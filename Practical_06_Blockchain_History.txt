PYTHON CODE

# Save as: practical6.py

import hashlib
import time

class Block:
    def __init__(self, data, previous_hash):
        self.timestamp = time.ctime()
        self.data = data
        self.previous_hash = previous_hash
        self.hash = self.calculate_hash()

    def calculate_hash(self):
        text = self.timestamp + self.data + self.previous_hash
        return hashlib.sha256(text.encode()).hexdigest()

# Create blocks
block1 = Block("A sends 10 coins to B", "0")
block2 = Block("B sends 5 coins to C", block1.hash)
block3 = Block("C sends 2 coins to D", block2.hash)

# Print blockchain
for block in [block1, block2, block3]:
    print("Time        :", block.timestamp)
    print("Transaction :", block.data)
    print("Hash        :", block.hash)
    print()
