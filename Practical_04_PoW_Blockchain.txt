# Save as: practical4.py

import hashlib
import time

# Block Class
class Block:
    def __init__(self, data, previous_hash):
        self.data = data
        self.previous_hash = previous_hash
        self.nonce = 0
        self.hash = self.mine_block()

    def calculate_hash(self):
        text = self.data + self.previous_hash + str(self.nonce)
        return hashlib.sha256(text.encode()).hexdigest()

    def mine_block(self):
        hash_value = self.calculate_hash()

        # PoW condition: hash starts with 0000
        while hash_value[:4] != "0000":
            self.nonce += 1
            hash_value = self.calculate_hash()

        return hash_value


# Create blockchain
genesis = Block("Genesis Block", "0")
block2 = Block("Transaction A -> B", genesis.hash)
block3 = Block("Transaction B -> C", block2.hash)

# Print blocks
print("Block 1 Hash :", genesis.hash)
print("Nonce        :", genesis.nonce)

print()

print("Block 2 Hash :", block2.hash)
print("Nonce        :", block2.nonce)

print()

print("Block 3 Hash :", block3.hash)
print("Nonce        :", block3.nonce)
