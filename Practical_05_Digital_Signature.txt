PYTHON CODE
-------------------------

# Save as: practical5.py

import hashlib

# Original document
document = "This is an important document"

# Sender creates signature using hash
signature = hashlib.sha256(document.encode()).hexdigest()

print("Original Document :", document)
print("Digital Signature :", signature)

# Receiver verifies document
received_document = document

verify_hash = hashlib.sha256(received_document.encode()).hexdigest()

if verify_hash == signature:
    print("Verification      : Valid Signature")
else:
    print("Verification      : Invalid Signature")
