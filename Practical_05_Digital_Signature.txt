# Save as: practical5.py

import hashlib

# User enters document
document = input("Enter Document Text : ")

# Create digital signature
signature = hashlib.sha256(document.encode()).hexdigest()

print("\nOriginal Document :", document)
print("Digital Signature :", signature)

# Receiver side verification
received_document = input("\nEnter Received Document For Verification : ")

verify_hash = hashlib.sha256(received_document.encode()).hexdigest()

if verify_hash == signature:
    print("Verification      : Valid Signature")
else:
    print("Verification      : Invalid Signature")
    print("Document Changed / Tampered")

    cout << endl;

    if(document == received_document)
    {
        cout << "Verification      : Valid Signature" << endl;
    }
    else
    {
        cout << "Verification      : Invalid Signature" << endl;
        cout << "Document Changed / Tampered" << endl;

}
